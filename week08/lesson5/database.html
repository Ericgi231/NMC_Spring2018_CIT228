Database Design

Customer: custID(PK), zipCode(FK), fName, lName, jobTitle, busPhone, mobPhone, faxNum, street, salesPerson(FK, employeeID), email, webpage

Employee: employID(PK), zipCode(FK), fName, lName, jobTitle, busPhone, mobPhone, homPhone, faxNum, email, startDate, salaried, wages, street

ZipCodes: zipCode(PK), city, state, country

MySQL Commands

Part #1

        CREATE DATABASE company;

Part #2

        USE company;

        CREATE TABLE Customer(
            custID INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
            fName VARCHAR(50),
            lName VARCHAR(50),
            jobTitle VARCHAR(50),
            busPhone VARCHAR(15),
            mobPhone VARCHAR(15),
            faxNum VARCHAR(25),
            street VARCHAR(50),
            zipCode INT,
            salesPerson INT,
            email VARCHAR(50),
            webpage VARCHAR(50)
        );
        
        CREATE TABLE Employee(
            employeeID INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
            fName VARCHAR(50),
            lName VARCHAR(50),
            jobTitle VARCHAR(50),
            busPhone VARCHAR(15),
            mobPhone VARCHAR(15),
            homPhone VARCHAR(15),
            faxNum VARCHAR(25),
            street VARCHAR(50),
            zipCode INT,
            email VARCHAR(50),
            webpage VARCHAR(50),
            startDate DATE,
            salaried BOOLEAN,
            wages INT
        );
        
        CREATE TABLE ZipCodes(
            zipCode INT NOT NULL PRIMARY KEY,
            city VARCHAR(50),
            state VARCHAR(50),
            country VARCHAR(50)
        );
